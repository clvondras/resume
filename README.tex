\newcommand{\sectiontitle}[1]{\begin{minipage}{\textwidth}\begin{center}\Large\MakeUppercase{#1}\end{center}\vspace{-5.5mm}\rule{\linewidth}{0.25mm}\vspace{2mm}\end{minipage}}

\newenvironment{tightressection}[1]{
  \begin{minipage}{\textwidth}
  \sectiontitle{#1}}
  {\end{minipage}}



% An entry for education
% Arg1: Institution
% Arg2: Program
% Arg3: Dates
\newcommand{\resedentry}[3]{
  \begin{minipage}{\textwidth}


  \textbf{#1}\\
  \textit{#2}\hspace{\stretch{1}}\textcolor{accent}{#3}\vspace{1.5mm}

  \end{minipage}
}

% An environment for a resume section
\newenvironment{ressection}[1]{\sectiontitle{#1}}

% Resentry defines a position
% Arg1: Position title
% Arg2: Company or Office---Company
% Arg3: Location
% Arg4: Date Range
\newcommand{\resentryheader}[4]{

\textbf{#1}\hspace{\stretch{1}}\textcolor{accent}{#3}

\textit{#2}\hfill\textcolor{accent}{#4}
}

% A resitem is a simple list element
\newcommand{\resitem}[1]{

\item \begin{flushleft} #1 \end{flushleft}
}

% This is a resume entry with a header and bullet items
\newenvironment{resentry}[4]{
\begin{minipage}{\textwidth}
\resentryheader{#1}{#2}{#3}{#4}
\begin{itemize}[noitemsep,nolistsep]
}{
\end{itemize}

\end{minipage}
}
